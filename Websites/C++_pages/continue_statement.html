Example of Continue; statement in use:

            
The continue; statement causes the loop to stop for the current iteration, 
but contiues the loop at the next iteration.
It's like break; but for one iteration only.

            
        #include <iostream>
        using namespace std;
        
        int main() {

            cout << "\n\n------------------------------------------------------\n";    
            cout << "\nLooping some numbers:\n";

            for (int i=0; i<10; i++) 
            {
                cout << "\ni => "  i;
            }

            cout << "\n\n------------------------------------------------------\n";    

            cout << "\nLooping some numbers with a continue statement to skip the mod 3's:\n";
            for (int i=0; i<10; i++) 
            {
                if ( i % 3 == 0 ) continue;
                cout << "\ni => " << i;
            }

            cout << "\n\n------------------------------------------------------\n\n";    

        }
